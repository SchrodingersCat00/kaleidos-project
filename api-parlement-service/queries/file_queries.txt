PREFIX nfo: <http://www.semanticdesktop.org/ontologies/2007/03/22/nfo#>

SELECT *
WHERE {
    GRAPH <http://mu.semte.ch/application> {
        ?subcase a dbpedia:UnitOfWork .
        ?subcase ext:bevatDocumentversie ?documentVersie .
        ?documentVersie ext:gekozenDocumentNaam ?docName .
        ?documentVersie ext:versieNummer ?versieNummer .
        ?documentVersie ext:file ?file .
        ?file dct:format ?fileFormat .
        ?file nfo:fileSize ?fileSize .
        ?file dbpedia:fileExtension ?fileExtension .
        ?documentVersie dct:created ?created .
        OPTIONAL {
            ?documentVersie ext:mededelingBevatDocumentversie ?announcement .
        }
    }
}
